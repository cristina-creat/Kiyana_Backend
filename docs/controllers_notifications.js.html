'use strict'

var Notification = require('../models/notification');

/***************************************
*
*
*	Notifications functions
*
*
***************************************/

exports.getNotifications = function(req,res){
	/**
	* Return a list of user Notifications
	*
	* returns Notifications
	**/

    var populate = [
		{path: '_user', select:'avatar firtname lastname'},
		{path: '_business_users._business_user', select:'avatar firstname lastname'}
    ];

	var filter = {};

	if ( req.user.business ) {
		filter = {
			'_business_users._business_user': req.user._id 
		}
	} else {
		filter = {
			'_user':req.user._id 
		}
	}

	Notification.find(filter).select('title message _event created_at _user').populate(populate).sort('-created_at').limit(20).exec(function(err, data){
		data = data.map( (el) => {
			el.message = el.message.substr(0, 40);
			return el;
		})
		res.send({data: data});
	});
}
